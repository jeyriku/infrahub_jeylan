! Serveur: jeysrv10
! IP: 192.168.0.237
! Communauté: jeyricorp
!
! ============================================================================
! Configuration ACL 18 actuelle
! ============================================================================

! IPs déjà autorisées:
access-list 18 permit 192.168.0.239
access-list 18 permit 192.168.0.248
access-list 18 permit 192.168.0.249

! ============================================================================
! Configuration à ajouter
! ============================================================================

! 1. Ajouter l'IP du serveur Infrahub à l'ACL SNMP existante
access-list 18 permit 192.168.0.237

! 2. Vérifier la configuration complète
show access-lists 18

! Résultat attendu:
! access-list 18 permit 192.168.0.237
! access-list 18 permit 192.168.0.239
! access-list 18 permit 192.168.0.248
! access-list 18 permit 192.168.0.249

! Résultat attendu:
! access-list 18 permit 192.168.0.237
! access-list 18 permit 192.168.0.239
! access-list 18 permit 192.168.0.248
! access-list 18 permit 192.168.0.249

! 3. Vérifier la communauté SNMP
show snmp community

! 4. Si la communauté n'existe pas encore, la créer avec l'ACL 18
snmp-server community jeyricorp RO 18

! 5. Vérifier que SNMP est activé
show snmp
